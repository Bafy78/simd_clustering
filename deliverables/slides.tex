\begin{frame}{Three algorithms, three computational structures}
  \vspace{-0.5em}
  \begingroup
  \fontsize{6.35}{7.10}\selectfont
  \setlength{\tabcolsep}{2.25pt}
  \renewcommand{\arraystretch}{1.10}
  \begin{tabular}{@{}>{\raggedright\arraybackslash}p{1.20cm}
                       >{\raggedright\arraybackslash}p{2.50cm}
                       >{\raggedright\arraybackslash}p{4.82cm}
                       >{\centering\arraybackslash}p{3.0cm}@{}}
    \toprule
    \textbf{Method} & \textbf{Output and specificities} & \textbf{Dominant structure and complexities} & \textbf{Visual} \\
    \midrule

    \textbf{K-Means}\par \textcolor{myuniversity}{\textbf{Lloyd}}
      & Hard labels; $K$ fixed. Centroid model: roughly convex, similarly
        sized clusters.
      & \textbf{Lloyd iteration:} $N\!\times\!K$ distances
        ($D$-reduction) $\rightarrow\arg\min_K$; then segmented sums and
        counts. Regular, sample-parallel assignment; less regular update
        writes.\par\smallskip
        \textcolor{myuniversity}{%
          $T=\mathcal O(TNKD)$ \quad
          $S=\mathcal O((N+K)D)$} \smallskip\smallskip\smallskip
      & \tablevisual{kmeans_visual.png} \\

    \midrule

    \textbf{Gaussian}\par \textbf{mixture}\par \textcolor{myuniversity}{\textbf{EM}}
      & Soft responsibilities and density model; $K$ fixed. Covariances
        represent elliptical clusters.
      & \textbf{EM:} E-step = Mahalanobis scores + exp/log-sum-exp; M-step =
        weighted moments. Full covariance adds outer products and $K$ matrix
        factorizations. Dense, regular, high I.\par\smallskip
        \textcolor{myuniversity}{%
          $T_{\mathrm{full}}/t=\mathcal O(NKD^2+KD^3)$;\quad
          $T_{\mathrm{diag/sph}}/t=\mathcal O(NKD)$}\par
        \textcolor{myuniversity}{%
          $S_{\mathrm{stream,full}}=\mathcal O(ND+KD^2)$;
          materialized responsibilities add $\mathcal O(NK)$} \smallskip
      & \tablevisual{gmm_full_covariance_visual.png} \\

    \midrule

    \textbf{HDBSCAN}
      & $k$ inferred from density/stability; non-convex, unequal-density
        clusters. Explicit noise, hierarchy and membership strengths.
      & \textbf{Pipeline:} pairwise/core distances $\rightarrow$ mutual-
        reachability MST $\rightarrow$ hierarchy $\rightarrow$ stability
        selection. Regular, reduction-heavy front end; dependent, irregular
        graph/tree tail.\par\smallskip
        \textcolor{myuniversity}{%
          $T=\mathcal O(DN^2)$ ;\quad $S=\mathcal O(N^2)$ with the dense matrix} \smallskip
      & \tablevisual{hdbscan_visual.png} \\

    \bottomrule
  \end{tabular}
  \endgroup

  \vspace{0.15em}
  \centering
  \tiny
  \textcolor{myuniversity}{$N$: samples \quad $D$: dimensions \quad $K$: clusters \quad $T$: iterations}
\end{frame}
